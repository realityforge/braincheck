001package org.realityforge.braincheck;
002
003import java.lang.reflect.Field;
004import java.util.function.Supplier;
005import javax.annotation.Nonnull;
006import javax.annotation.Nullable;
007
008/**
009 * Utility class for interacting with BrainCheck in tests.
010 */
011@GwtIncompatible
012public final class BrainCheckTestUtil
013{
014  private BrainCheckTestUtil()
015  {
016  }
017
018  /**
019   * The type of the guard.
020   * This is only used during development process and will be optimized out in production mode.
021   */
022  @GwtIncompatible
023  public enum GuardType
024  {
025    INVARIANT,
026    API_INVARIANT
027  }
028
029  /**
030   * Interface used to receive details about an invoked guard.
031   * This is only used internally during development to collect the guards/invariants/etc to ensure
032   * that they comply with patterns and to ensure they are documented appropriately.
033   */
034  @GwtIncompatible
035  public interface OnGuardListener
036  {
037    void onGuard( @Nonnull GuardType type, @Nonnull String message, @Nonnull StackTraceElement[] stackTrace );
038  }
039
040  /**
041   * Specify a callback that is invoked anytime the {@link Guards#invariant(Supplier, Supplier)} method or
042   * the {@link Guards#apiInvariant(Supplier, Supplier)} method is invoked while in development mode.
043   *
044   * @param onGuardListener the listener.
045   */
046  public static void setOnGuardListener( @Nullable final OnGuardListener onGuardListener )
047  {
048    if ( null == onGuardListener )
049    {
050      Guards.setOnGuardListener( null );
051    }
052    else
053    {
054      final Guards.OnGuardListener listener =
055        ( type, message, stackTrace ) -> onGuardListener.onGuard( GuardType.valueOf( type.name() ),
056                                                                  message,
057                                                                  stackTrace );
058      Guards.setOnGuardListener( listener );
059    }
060  }
061
062  /**
063   * Reset the state of BrainCheck config to either production or development state.
064   *
065   * @param productionMode true to set it to production mode configuration, false to set it to development mode config.
066   */
067  public static void resetConfig( final boolean productionMode )
068  {
069    if ( BrainCheckConfig.isProductionEnvironment() )
070    {
071      throw new IllegalStateException( "Unable to reset config as BrainCheck is in production mode" );
072    }
073
074    if ( productionMode )
075    {
076      setVerboseErrorMessages( false );
077      setCheckInvariants( false );
078      setCheckApiInvariants( false );
079    }
080    else
081    {
082      setVerboseErrorMessages( true );
083      setCheckInvariants( true );
084      setCheckApiInvariants( true );
085    }
086    setOnGuardListener( null );
087  }
088
089  /**
090   * Configure the verbose error messages setting.
091   *
092   * @param verboseErrorMessages the verbose error messages setting.
093   */
094  public static void setVerboseErrorMessages( final boolean verboseErrorMessages )
095  {
096    setConstant( "VERBOSE_ERROR_MESSAGES", verboseErrorMessages );
097  }
098
099  /**
100   * Configure the "check invariants" setting.
101   *
102   * @param checkInvariants the "check invariants" setting.
103   */
104  public static void setCheckInvariants( final boolean checkInvariants )
105  {
106    setConstant( "CHECK_INVARIANTS", checkInvariants );
107  }
108
109  /**
110   * Configure the "check api invariants" setting.
111   *
112   * @param checkApiInvariants the "check api invariants" setting.
113   */
114  public static void setCheckApiInvariants( final boolean checkApiInvariants )
115  {
116    setConstant( "CHECK_API_INVARIANTS", checkApiInvariants );
117  }
118
119  /**
120   * Set the specified field name on BrainCheckConfig.
121   */
122  @SuppressWarnings( "NonJREEmulationClassesInClientCode" )
123  private static void setConstant( @Nonnull final String fieldName, final boolean value )
124  {
125    try
126    {
127      final Field field = BrainCheckConfig.class.getDeclaredField( fieldName );
128      field.setAccessible( true );
129      field.set( null, value );
130    }
131    catch ( NoSuchFieldException | IllegalAccessException e )
132    {
133      throw new IllegalStateException( "Unable to change constant " + fieldName, e );
134    }
135  }
136}
