029/**
030 * This class is used by test infrastructure to collect and match all invocations invariant checks
031 * in the {@link Guards} class. This class will be invoked any time {@link Guards#invariant(Supplier, Supplier)}
032 * is invoked or {@link Guards#apiInvariant(Supplier, Supplier)} is invoked and will receive the message. If
033 * the message matches the pattern <code>"[Key]-####: [Message]"</code> and this class has been configured to
034 * match <code>Key</code> message then this message will be recorded or matched against predefined message template.
035 *
036 * <p>The message templates are recorded in a json file. The json contains an array of entries of the
037 * form belows. The <code>messagePattern</code> property can also include the value <code>%s</code> which
038 * is replaced with <code>.*</code> which can be replaced by any string during matching.</p>
039 *
040 * <pre>
041 *     {
042 *     "code": 1,
043 *     "type": "API_INVARIANT",
044 *     "messagePattern": "Invoked Arez.createZone() but zones are not enabled.",
045 *     "callers": [
046 *       {
047 *         "class": "arez.Arez",
048 *         "method": "createZone",
049 *         "file": "Arez.java",
050 *         "lineNumber": 185
051 *       }
052 *     ]
053 *   }
054 * </pre>
055 *
056 * <p>The class makes a few assumptions. It assumes that every invariant message with the specified
057 * <code>key</code> is invoked while the collector is active and thus if a message does not occur that
058 * it no longer exists. It also assumes that only invariant messages that match the above pattern are
059 * of interest.</p>
060 *
061 * <p>To enable this class it is typical to hook it into appropriate places according to your test
062 * framework of choice. In TestNG the base test class may look like:</p>
063 *
064 * <pre>
065 * public abstract class AbstractTest
066 * {
067 *   private static final GuardMessageCollector c_messages = createCollector();
068 *
069 *   {@code @}BeforeSuite
070 *   protected void beforeSuite() { c_messages.onTestSuiteStart(); }
071 *
072 *   {@code @}BeforeMethod
073 *   protected void beforeTest() { c_messages.onTestStart(); }
074 *
075 *   {@code @}AfterMethod
076 *   protected void afterTest() { c_messages.onTestComplete(); }
077 *
078 *   {@code @}AfterSuite
079 *   protected void afterSuite() { c_messages.onTestSuiteComplete(); }
080 *
081 *   private static GuardMessageCollector createCollector() { ... }
082 * }
083 * </pre>
084 *
085 * <p>The primary purpose of the class is to enable the validation and collection of invariant checks
086 * for the purpose of documentation. The output file includes the invariant pattern as well as the
087 * location in the source file where the invariant checks are generated. This makes it easy to
088 * generate errors with cross references as required.</p>
089 */
090@GwtIncompatible
091public final class GuardMessageCollector
092{
093  @Nonnull
094  private final Map<Integer, Message> _messages = new HashMap<>();
095  @Nonnull
096  private final String _key;
097  @Nonnull
098  private final File _file;
099  private final boolean _saveIfChanged;
100  private final boolean _deleteIfUnmatched;
101  private final boolean _recordCallers;
102  private long _loadTime;
103  private int _matchFailureCount;

154                                @Nonnull final File file,

157                                final boolean recordCallers )
158  {
159    _key = Objects.requireNonNull( key );
160    _file = Objects.requireNonNull( file );
161    _saveIfChanged = saveIfChanged;
162    _deleteIfUnmatched = deleteIfUnmatched;
163    _recordCallers = recordCallers;

171  /**
172   * Hook method that should be invoked before any test starts.
173   * This method will load the message templates if the file exists otherwise it will just reset internal state.
174   */
175  public void onTestSuiteStart()
176  {
177    _matchFailureCount = 0;
178    loadIfRequired();
179  }

181  /**
182   * Hook method that should be invoked before a test starts.
183   */
184  public void onTestStart()
185  {
186    BrainCheckTestUtil.setOnGuardListener( this::onGuardInvoked );
187  }

189  /**
190   * Hook method that should be invoked after a test completes.
191   */
192  public void onTestComplete()
193  {
194    BrainCheckTestUtil.setOnGuardListener( null );
195  }

197  /**
198   * Hook method that should be invoked after all tests have completed.
199   * If the expected invariant invocations no longer match the actual invariant invocations
200   * then this method will either save the new message template or generate an exception depending
201   * on the value of the <code>saveIfChanged</code> used to create this class.
202   *
203   * @param suiteSuccessful true if test suite completed with no failures, false otherwise.
204   */
205  public void onTestSuiteComplete( final boolean suiteSuccessful )
206  {
207    if ( needsSave() )
208    {
209      if ( _saveIfChanged )
210      {
211        save( suiteSuccessful );
212      }
213      else
214      {
215        final List<Message> unsavedMessages =
216          _messages.values().stream().filter( m -> m.needsSave( _recordCallers ) ).collect( Collectors.toList() );
217        throw new IllegalStateException( "Diagnostic messages template is out of date. " + unsavedMessages.size() +
218                                         " messages need to be updated including messages:\n" +
219                                         unsavedMessages.stream()
220                                           .map( Message::toString )
221                                           .collect( Collectors.joining( "\n" ) ) );
222      }
223    }
224  }

226  private void onGuardInvoked( @Nonnull final BrainCheckTestUtil.GuardType type,
227                               @Nonnull final String message,
228                               @Nonnull final StackTraceElement[] stackTrace )
229  {
230    final Matcher matcher = Pattern.compile( "^" + _key + "-(\\d\\d\\d\\d): (.*)$" ).matcher( message );
231    if ( matcher.matches() )
232    {
233      final int code = Integer.parseInt( matcher.group( 1 ) );
234      final String msg = matcher.group( 2 );
235
236      matchOrRecordDiagnosticMessage( code, type, msg, stackTrace[ 0 ] );
237    }
238  }
239
240  private void loadIfRequired()
241  {
242    if ( _file.exists() )
243    {
244      final long lastModified = _file.lastModified();
245      if ( _loadTime != lastModified )
246      {
247        loadMessages();
248        _loadTime = lastModified;
249      }
250    }
251    else
252    {
253      _messages.clear();
254      _loadTime = 0;
255    }
256  }
257
258  private void loadMessages()
259  {
260    _messages.clear();
261    try ( final FileInputStream inputStream = new FileInputStream( _file ) )
262    {
263      final JsonReader reader = Json.createReader( inputStream );
264      final JsonArray top = reader.readArray();
265      final int size = top.size();
266      for ( int i = 0; i < size; i++ )
267      {
268        final JsonObject entry = top.getJsonObject( i );
269        final int code = entry.getInt( "code" );
270        final BrainCheckTestUtil.GuardType type = BrainCheckTestUtil.GuardType.valueOf( entry.getString( "type" ) );
271        final String messagePattern = entry.getString( "messagePattern" );
272        if ( _messages.containsKey( code ) )
273        {
274          throw new IllegalStateException( "Failed to load diagnostic messages file " + _file + " as it is " +
275                                           "incorrectly formatted with duplicate entries for code " + code );
276        }
277        final HashSet<StackTraceElement> callers = new HashSet<>();
278        if ( entry.containsKey( "callers" ) )
279        {
280          final JsonArray callersData = entry.getJsonArray( "callers" );
281          final int callerCount = callersData.size();
282          for ( int j = 0; j < callerCount; j++ )
283          {
284            final JsonObject callerData = callersData.getJsonObject( j );
285            final String className = callerData.getString( "class" );
286            final String methodName = callerData.getString( "method" );
287            final String fileName = callerData.getString( "file" );
288            final int lineNumber = callerData.getInt( "lineNumber" );
289            callers.add( new StackTraceElement( className, methodName, fileName, lineNumber ) );
290          }
291        }
292        _messages.put( code, new Message( _key, code, type, messagePattern, false, callers ) );
293      }
294    }
295    catch ( final IOException ioe )
296    {
297      throw new IllegalStateException( "Failed to read diagnostic messages file " + _file + ".", ioe );
298    }
299  }

301  private void save( final boolean suiteSuccessful )
302  {
303    final Map<String, Object> properties = new HashMap<>();
304    properties.put( JsonGenerator.PRETTY_PRINTING, true );
305
306    final JsonGeneratorFactory generatorFactory = Json.createGeneratorFactory( properties );
307    try
308    {
309      try ( final FileOutputStream output = new FileOutputStream( _file ) )
310      {
311        final JsonGenerator g = generatorFactory.createGenerator( output );
312        g.writeStartArray();
313        _messages.values()
314          .stream()
315          // Avoid deleting an entry if we failed matching some invariants as it is possible
316          // that some invariant checks were never reached during test run
317          .filter( m -> !_deleteIfUnmatched || !suiteSuccessful || !m.getCallers().isEmpty() )
318          .sorted( Comparator.comparingInt( Message::getCode ) )
319          .forEachOrdered( m -> {
320            g.writeStartObject();
321            g.write( "code", m.getCode() );
322            g.write( "type", m.getType().name() );
323            g.write( "messagePattern", m.getMessagePattern() );
324            if ( _recordCallers )
325            {
326              g.writeStartArray( "callers" );
327              final StackTraceElement[] callers =
328                m.getCallers().stream().sorted( this::compareElements ).toArray( StackTraceElement[]::new );
329              for ( final StackTraceElement caller : callers )
330              {
331                g.writeStartObject();
332                g.write( "class", caller.getClassName() );
333                g.write( "method", caller.getMethodName() );
334                g.write( "file", caller.getFileName() );
335                g.write( "lineNumber", caller.getLineNumber() );
336                g.writeEnd();
337              }
338
339              g.writeEnd();
340            }
341            g.writeEnd();
342          } );
343        g.writeEnd();
344        g.close();
345      }
346      formatJson( _file );
347    }
348    catch ( final IOException ioe )
349    {
350      throw new IllegalStateException( "Failed to write diagnostic messages file " + _file + ".", ioe );
351    }
352  }
353
354  private int compareElements( @Nonnull final StackTraceElement o1, @Nonnull final StackTraceElement o2 )
355  {
356    final int v = Objects.compare( o1.getClassName(), o2.getClassName(), String::compareTo );
357    if ( 0 != v )
358    {
359      return v;
360    }
361    else
362    {
363      return Objects.compare( o1.getLineNumber(), o2.getLineNumber(), Integer::compareTo );
364    }
365  }
366
367  private boolean needsSave()
368  {
369    return _messages.values().stream().anyMatch( m -> m.needsSave( _recordCallers ) );
370  }

372  /**
373   * Format the json file.
374   * This is horribly inefficient but it is not called very often so ... meh.
375   */
376  private void formatJson( @Nonnull final File file )
377    throws IOException
378  {
379    final byte[] data = Files.readAllBytes( file.toPath() );
380    final String jsonData = new String( data, StandardCharsets.UTF_8 );
381
382    final String output =
383      jsonData
384        .replaceAll( "(?m)^ {4}\\{", "  {" )
385        .replaceAll( "(?m)^ {4}}", "  }" )
386        .replaceAll( "(?m)^ {8}\"", "    \"" )
387        .replaceAll( "(?m)^ {8}]", "    ]" )
388        .replaceAll( "(?m)^ {12}\\{", "      {" )
389        .replaceAll( "(?m)^ {12}}", "      }" )
390        .replaceAll( "(?m)^ {16}\"", "        \"" )
391        .replaceAll( "(?m)^\n\\[\n", "[\n" ) +
392      "\n";
393    Files.write( file.toPath(), output.getBytes( StandardCharsets.UTF_8 ) );
394  }

396  private void recordDiagnosticMessage( final int code,
397                                        @Nonnull final BrainCheckTestUtil.GuardType type,
398                                        @Nonnull final String messagePattern,
399                                        @Nonnull final StackTraceElement caller )
400  {
401    final Message message = new Message( _key, code, type, messagePattern, true, new HashSet<>() );
402    message.recordCaller( caller );
403    _messages.put( code, message );
404  }

406  private void matchOrRecordDiagnosticMessage( final int code,
407                                               @Nonnull final BrainCheckTestUtil.GuardType type,
408                                               @Nonnull final String message,
409                                               @Nonnull final StackTraceElement caller )
410  {
411    final Message m = _messages.get( code );
412    if ( null == m )
413    {
414      recordDiagnosticMessage( code, type, message, caller );
415    }
416    else
417    {
418      m.recordCaller( caller );
419      final StringBuilder sb = new StringBuilder();
420
421      final String messagePattern = m.getMessagePattern();
422      int lastOffset = 0;
423      int offset;
424      while ( -1 != ( offset = messagePattern.indexOf( "%s", lastOffset ) ) )
425      {
426        final String segment = messagePattern.substring( lastOffset, offset );
427        sb.append( Pattern.quote( segment ) );
428        sb.append( ".*" );
429        lastOffset = offset + 2;
430      }
431      sb.append( Pattern.quote( messagePattern.substring( lastOffset ) ) );
432
433      final Pattern pattern = Pattern.compile( sb.toString() );
434      if ( !pattern.matcher( message ).matches() )
435      {
436        _matchFailureCount++;
437        throw new AssertionError( "Failed to match " + type + " diagnostic message with " +
438                                  "key " + _key + " and code " + code + ".\n" +
439                                  "Expected pattern:\n" + messagePattern + "\n\n" +
440                                  "Actual Message:\n" + message );
441      }
442      else if ( !Objects.equals( m.getType(), type ) )
443      {
444        _matchFailureCount++;
445        throw new AssertionError( "Failed to match diagnostic message type with " +
446                                  "key " + _key + " and code " + code + "." );
447      }
448    }
449  }

451  /**
452   * Class representing a message of a single code.
453   */
454  @GwtIncompatible
455  private static final class Message
456  {
457    @Nonnull
458    private final String _key;
459    private final int _code;
460    @Nonnull
461    private final BrainCheckTestUtil.GuardType _type;
462    @Nonnull
463    private final String _messagePattern;
464    private final boolean _needsSave;
465    private final Set<StackTraceElement> _originalCallers = new HashSet<>();
466    private final Set<StackTraceElement> _callers = new HashSet<>();
467
468    Message( @Nonnull final String key,
469             final int code,
470             @Nonnull final BrainCheckTestUtil.GuardType type,
471             @Nonnull final String messagePattern,
472             final boolean needsSave,
473             @Nonnull final Set<StackTraceElement> callers )
474    {
475      _key = Objects.requireNonNull( key );
476      _code = code;
477      _type = Objects.requireNonNull( type );
478      _messagePattern = Objects.requireNonNull( messagePattern );
479      _needsSave = needsSave;
480      _originalCallers.addAll( Objects.requireNonNull( callers ) );
481    }
482
483    int getCode()
484    {
485      return _code;
486    }
487
488    @Nonnull
489    BrainCheckTestUtil.GuardType getType()
490    {
491      return _type;
492    }
493
494    @Nonnull
495    String getMessagePattern()
496    {
497      return _messagePattern;
498    }
499
500    boolean needsSave( final boolean recordCallers )
501    {
502      return _needsSave ||
503             ( recordCallers && !Objects.equals( _originalCallers, _callers ) ) ||
504             _callers.isEmpty();
505    }
506
507    void recordCaller( @Nonnull final StackTraceElement caller )
508    {
509      _callers.add( caller );
510    }
511
512    @Nonnull
513    Set<StackTraceElement> getCallers()
514    {
515      return Collections.unmodifiableSet( _callers );
516    }
517
518    @Override
519    public String toString()
520    {
521      return _key + "-" + String.format( "%4d", getCode() ) + ": " + getMessagePattern();
522    }
523  }
