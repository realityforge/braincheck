001package org.realityforge.braincheck;
002
003import java.util.function.Supplier;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006
007/**
008 * A utility class used to perform assertions and invariant checks.
009 */
010public final class Guards
011{
012  @Nullable
013  private static OnGuardListener c_onGuardListener;
014
015  /**
016   * The type of the guard.
017   * This is only used during development process and will be optimized out in production mode.
018   */
019  enum Type
020  {
021    FAIL,
022    INVARIANT,
023    API_INVARIANT
024  }
025
026  /**
027   * Interface used to receive details about an invoked guard.
028   * This is only used internally during development to collect the guards/invariants/etc to ensure
029   * that they comply with patterns and to ensure they are documented appropriately.
030   */
031  interface OnGuardListener
032  {
033    void onGuard( @Nonnull Type type, @Nonnull String message, @Nonnull StackTraceElement[] stackTrace );
034  }
035
036  static void setOnGuardListener( @Nullable final OnGuardListener onGuardListener )
037  {
038    assert BrainCheckConfig.isDevelopmentEnvironment();
039    c_onGuardListener = onGuardListener;
040  }
041
042  private Guards()
043  {
044  }
045
046  /**
047   * Check an api invariant in the code base.
048   * The invariant condition should return true if the library user is using the api correctly. This invariant
049   * should be disabled in production environments but users of the library may choose to enable it when using
050   * the library.
051   *
052   * <p>If the condition is false then an {@link IllegalStateException} is thrown.
053   * The invariant check will be skipped unless the configuration setting {@link BrainCheckConfig#checkInvariants()}
054   * is true. A null message is used rather than supplied message unless {@link BrainCheckConfig#verboseErrorMessages()}
055   * is true.</p>
056   *
057   * @param condition the condition to check.
058   * @param message   the message supplier used if verbose messages enabled.
059   * @throws IllegalStateException if condition returns false.
060   */
061  public static void apiInvariant( @Nonnull final Supplier<Boolean> condition,
062                                   @Nonnull final Supplier<String> message )
063  {
064    if ( BrainCheckConfig.isDevelopmentEnvironment() && null != c_onGuardListener )
065    {
066      c_onGuardListener.onGuard( Type.API_INVARIANT,
067                                 BrainCheckUtil.safeGetString( message ),
068                                 StackTraceUtil.getStackTrace( 2 ) );
069    }
070    if ( BrainCheckConfig.checkApiInvariants() )
071    {
072      boolean conditionResult = isConditionTrue( condition, message );
073      if ( !conditionResult )
074      {
075        doFail( message );
076      }
077    }
078  }
079
080  /**
081   * Check an invariant in code base.
082   * The invariant condition should always return true. Returning false indicates that the host application
083   * or library has an unexpected bug. These invariant checks may be compute intensive as they will likely
084   * be disabled in production environments.
085   *
086   * <p>If the condition is false then an {@link IllegalStateException} is thrown.
087   * The invariant check will be skipped unless the configuration setting {@link BrainCheckConfig#checkInvariants()}
088   * is true. A null message is used rather than supplied message unless {@link BrainCheckConfig#verboseErrorMessages()}
089   * is true.</p>
090   *
091   * @param condition the condition to check.
092   * @param message   the message supplier used if verbose messages enabled.
093   * @throws IllegalStateException if condition returns false.
094   */
095  public static void invariant( @Nonnull final Supplier<Boolean> condition,
096                                @Nonnull final Supplier<String> message )
097  {
098    if ( BrainCheckConfig.isDevelopmentEnvironment() && null != c_onGuardListener )
099    {
100      c_onGuardListener.onGuard( Type.INVARIANT,
101                                 BrainCheckUtil.safeGetString( message ),
102                                 StackTraceUtil.getStackTrace( 2 ) );
103    }
104    if ( BrainCheckConfig.checkInvariants() )
105    {
106      if ( !isConditionTrue( condition, message ) )
107      {
108        doFail( message );
109      }
110    }
111  }
112
113  /**
114   * Return the result of specified condition.
115   * If there is an error resolving condition then fail or return false depending on config settings.
116   *
117   * @return the result of specified condition.
118   */
119  private static boolean isConditionTrue( @Nonnull final Supplier<Boolean> condition,
120                                          @Nonnull final Supplier<String> message )
121  {
122    try
123    {
124      return condition.get();
125    }
126    catch ( final Throwable t )
127    {
128      doFail( () -> "Error checking condition.\n" +
129                    "Message: " + BrainCheckUtil.safeGetString( message ) + "\n" +
130                    "Throwable:\n" + BrainCheckUtil.throwableToString( t ) );
131    }
132    return false;
133  }
134
135  /**
136   * Throw an IllegalStateException with supplied detail message.
137   * The exception is not thrown unless {@link BrainCheckConfig#checkInvariants()} is true.
138   * The exception will ignore the supplied message unless {@link BrainCheckConfig#verboseErrorMessages()} is true.
139   *
140   * @param message the message supplier used if verbose messages enabled.
141   * @throws IllegalStateException when called.
142   */
143  public static void fail( @Nonnull final Supplier<String> message )
144  {
145    if ( BrainCheckConfig.isDevelopmentEnvironment() && null != c_onGuardListener )
146    {
147      c_onGuardListener.onGuard( Type.FAIL,
148                                 BrainCheckUtil.safeGetString( message ),
149                                 StackTraceUtil.getStackTrace( 2 ) );
150    }
151    if ( BrainCheckConfig.checkInvariants() )
152    {
153      doFail( message );
154    }
155  }
156
157  private static void doFail( @Nonnull final Supplier<String> message )
158  {
159    /*
160     * This flag will only be present and set when GWT is compiling the source code and the relevant
161     * compile time property is defined. Thus this will be false in normal jre runtime environment.
162     */
163    if ( "ENABLED".equals( System.getProperty( "jre.debugMode" ) ) )
164    {
165      DebuggerUtil.debugger();
166    }
167    if ( BrainCheckConfig.verboseErrorMessages() )
168    {
169      throw new IllegalStateException( BrainCheckUtil.safeGetString( message ) );
170    }
171    else
172    {
173      throw new IllegalStateException();
174    }
175  }
176}
