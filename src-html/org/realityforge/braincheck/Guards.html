001package org.realityforge.braincheck;
002
003import java.util.function.Supplier;
004import javax.annotation.Nonnull;
005
006/**
007 * A utility class used to perform assertions and invariant checks.
008 */
009public final class Guards
010{
011  private Guards()
012  {
013  }
014
015  /**
016   * Check an api invariant in the code base.
017   * The invariant condition should return true if the library user is using the api correctly. This invariant
018   * should be disabled in production environments but users of the library may choose to enable it when using
019   * the library.
020   *
021   * <p>If the condition is false then an {@link IllegalStateException} is thrown.
022   * The invariant check will be skipped unless the configuration setting {@link BrainCheckConfig#checkInvariants()}
023   * is true. A null message is used rather than supplied message unless {@link BrainCheckConfig#verboseErrorMessages()}
024   * is true.</p>
025   *
026   * @param condition the condition to check.
027   * @param message   the message supplier used if verbose messages enabled.
028   * @throws IllegalStateException if condition returns false.
029   */
030  public static void apiInvariant( @Nonnull final Supplier<Boolean> condition,
031                                   @Nonnull final Supplier<String> message )
032  {
033    if ( BrainCheckConfig.checkApiInvariants() )
034    {
035      boolean conditionResult = isConditionTrue( condition, message );
036      if ( !conditionResult )
037      {
038        doFail( message );
039      }
040    }
041  }
042
043  /**
044   * Check an invariant in code base.
045   * The invariant condition should always return true. Returning false indicates that the host application
046   * or library has an unexpected bug. These invariant checks may be compute intensive as they will likely
047   * be disabled in production environments.
048   *
049   * <p>If the condition is false then an {@link IllegalStateException} is thrown.
050   * The invariant check will be skipped unless the configuration setting {@link BrainCheckConfig#checkInvariants()}
051   * is true. A null message is used rather than supplied message unless {@link BrainCheckConfig#verboseErrorMessages()}
052   * is true.</p>
053   *
054   * @param condition the condition to check.
055   * @param message   the message supplier used if verbose messages enabled.
056   * @throws IllegalStateException if condition returns false.
057   */
058  public static void invariant( @Nonnull final Supplier<Boolean> condition,
059                                @Nonnull final Supplier<String> message )
060  {
061    if ( BrainCheckConfig.checkInvariants() )
062    {
063      if ( !isConditionTrue( condition, message ) )
064      {
065        doFail( message );
066      }
067    }
068  }
069
070  /**
071   * Return the result of specified condition.
072   * If there is an error resolving condition then fail or return false depending on config settings.
073   *
074   * @return the result of specified condition.
075   */
076  private static boolean isConditionTrue( @Nonnull final Supplier<Boolean> condition,
077                                          @Nonnull final Supplier<String> message )
078  {
079    try
080    {
081      return condition.get();
082    }
083    catch ( final Throwable t )
084    {
085      doFail( () -> "Error checking condition.\n" +
086                    "Message: " + BrainCheckUtil.safeGetString( message ) + "\n" +
087                    "Throwable:\n" + BrainCheckUtil.throwableToString( t ) );
088    }
089    return false;
090  }
091
092  /**
093   * Throw an IllegalStateException with supplied detail message.
094   * The exception is not thrown unless {@link BrainCheckConfig#checkInvariants()} is true.
095   * The exception will ignore the supplied message unless {@link BrainCheckConfig#verboseErrorMessages()} is true.
096   *
097   * @param message the message supplier used if verbose messages enabled.
098   * @throws IllegalStateException when called.
099   */
100  public static void fail( @Nonnull final Supplier<String> message )
101  {
102    if ( BrainCheckConfig.checkInvariants() )
103    {
104      doFail( message );
105    }
106  }
107
108  private static void doFail( @Nonnull final Supplier<String> message )
109  {
110    /*
111     * This flag will only be present and set when GWT is compiling the source code and the relevant
112     * compile time property is defined. Thus this will be false in normal jre runtime environment.
113     */
114    if ( "ENABLED".equals( System.getProperty( "jre.debugMode" ) ) )
115    {
116      DebuggerUtil.debugger();
117    }
118    if ( BrainCheckConfig.verboseErrorMessages() )
119    {
120      throw new IllegalStateException( BrainCheckUtil.safeGetString( message ) );
121    }
122    else
123    {
124      throw new IllegalStateException();
125    }
126  }
127}
