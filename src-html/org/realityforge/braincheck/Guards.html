001package org.realityforge.braincheck;
002
003import java.util.function.Supplier;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006
007/**
008 * A utility class used to perform assertions and invariant checks.
009 */
010public final class Guards
011{
012  @Nullable
013  private static OnGuardListener c_onGuardListener;
014
015  /**
016   * The type of the guard.
017   * This is only used during development process and will be optimized out in production mode.
018   */
019  enum Type
020  {
021    INVARIANT,
022    API_INVARIANT
023  }
024
025  /**
026   * Interface used to receive details about an invoked guard.
027   * This is only used internally during development to collect the guards/invariants/etc to ensure
028   * that they comply with patterns and to ensure they are documented appropriately.
029   */
030  interface OnGuardListener
031  {
032    void onGuard( @Nonnull Type type, @Nonnull String message, @Nonnull StackTraceElement[] stackTrace );
033  }
034
035  static void setOnGuardListener( @Nullable final OnGuardListener onGuardListener )
036  {
037    assert BrainCheckConfig.isDevelopmentEnvironment();
038    c_onGuardListener = onGuardListener;
039  }
040
041  private Guards()
042  {
043  }
044
045  /**
046   * Check an api invariant in the code base.
047   * The invariant condition should return true if the library user is using the api correctly. This invariant
048   * should be disabled in production environments but users of the library may choose to enable it when using
049   * the library.
050   *
051   * <p>If the condition is false then an {@link IllegalStateException} is thrown.
052   * The invariant check will be skipped unless the configuration setting {@link BrainCheckConfig#checkInvariants()}
053   * is true. A null message is used rather than supplied message unless {@link BrainCheckConfig#verboseErrorMessages()}
054   * is true.</p>
055   *
056   * @param condition the condition to check.
057   * @param message   the message supplier used if verbose messages enabled.
058   * @throws IllegalStateException if condition returns false.
059   */
060  public static void apiInvariant( @Nonnull final Supplier<Boolean> condition,
061                                   @Nonnull final Supplier<String> message )
062  {
063    if ( BrainCheckConfig.isDevelopmentEnvironment() && null != c_onGuardListener )
064    {
065      c_onGuardListener.onGuard( Type.API_INVARIANT,
066                                 BrainCheckUtil.safeGetString( message ),
067                                 StackTraceUtil.getStackTrace( 2 ) );
068    }
069    if ( BrainCheckConfig.checkApiInvariants() )
070    {
071      boolean conditionResult = isConditionTrue( condition, message );
072      if ( !conditionResult )
073      {
074        doFail( message );
075      }
076    }
077  }
078
079  /**
080   * Check an invariant in code base.
081   * The invariant condition should always return true. Returning false indicates that the host application
082   * or library has an unexpected bug. These invariant checks may be compute intensive as they will likely
083   * be disabled in production environments.
084   *
085   * <p>If the condition is false then an {@link IllegalStateException} is thrown.
086   * The invariant check will be skipped unless the configuration setting {@link BrainCheckConfig#checkInvariants()}
087   * is true. A null message is used rather than supplied message unless {@link BrainCheckConfig#verboseErrorMessages()}
088   * is true.</p>
089   *
090   * @param condition the condition to check.
091   * @param message   the message supplier used if verbose messages enabled.
092   * @throws IllegalStateException if condition returns false.
093   */
094  public static void invariant( @Nonnull final Supplier<Boolean> condition,
095                                @Nonnull final Supplier<String> message )
096  {
097    if ( BrainCheckConfig.isDevelopmentEnvironment() && null != c_onGuardListener )
098    {
099      c_onGuardListener.onGuard( Type.INVARIANT,
100                                 BrainCheckUtil.safeGetString( message ),
101                                 StackTraceUtil.getStackTrace( 2 ) );
102    }
103    if ( BrainCheckConfig.checkInvariants() )
104    {
105      if ( !isConditionTrue( condition, message ) )
106      {
107        doFail( message );
108      }
109    }
110  }
111
112  /**
113   * Return the result of specified condition.
114   * If there is an error resolving condition then fail or return false depending on config settings.
115   *
116   * @return the result of specified condition.
117   */
118  private static boolean isConditionTrue( @Nonnull final Supplier<Boolean> condition,
119                                          @Nonnull final Supplier<String> message )
120  {
121    try
122    {
123      return condition.get();
124    }
125    catch ( final Throwable t )
126    {
127      doFail( () -> "Error checking condition.\n" +
128                    "Message: " + BrainCheckUtil.safeGetString( message ) + "\n" +
129                    "Throwable:\n" + BrainCheckUtil.throwableToString( t ) );
130    }
131    return false;
132  }
133
134  /**
135   * Throw an IllegalStateException with supplied detail message.
136   * The exception is not thrown unless {@link BrainCheckConfig#checkInvariants()} is true.
137   * The exception will ignore the supplied message unless {@link BrainCheckConfig#verboseErrorMessages()} is true.
138   *
139   * @param message the message supplier used if verbose messages enabled.
140   * @throws IllegalStateException when called.
141   */
142  public static void fail( @Nonnull final Supplier<String> message )
143  {
144    if ( BrainCheckConfig.checkInvariants() )
145    {
146      doFail( message );
147    }
148  }
149
150  private static void doFail( @Nonnull final Supplier<String> message )
151  {
152    /*
153     * This flag will only be present and set when GWT is compiling the source code and the relevant
154     * compile time property is defined. Thus this will be false in normal jre runtime environment.
155     */
156    if ( "ENABLED".equals( System.getProperty( "jre.debugMode" ) ) )
157    {
158      DebuggerUtil.debugger();
159    }
160    if ( BrainCheckConfig.verboseErrorMessages() )
161    {
162      throw new IllegalStateException( BrainCheckUtil.safeGetString( message ) );
163    }
164    else
165    {
166      throw new IllegalStateException();
167    }
168  }
169}
