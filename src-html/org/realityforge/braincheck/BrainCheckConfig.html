001package org.realityforge.braincheck;
002
003/**
004 * Location of all compile time configuration settings for the toolkit.
005 */
006@SuppressWarnings( "StringEquality" )
007public final class BrainCheckConfig
008{
009  /**
010   * The provider abstraction is needed due to limitations in the way J2CL performs dead-code elimination.
011   * In J2CL the code <code>"production" == System.getProperty( "braincheck.environment" )</code> will return
012   * true when values match while this will not work in JRE mode but will work with later versions of GWT2.x
013   * it seems. Using a pair of classes with @GwtIncompatible allows us to have the best of both worlds.
014   */
015  private static final ConfigProvider PROVIDER = new ConfigProvider();
016
017  private static final boolean PRODUCTION_ENVIRONMENT = PROVIDER.isProductionEnvironment();
018  private static boolean VERBOSE_ERROR_MESSAGES = PROVIDER.verboseErrorMessages();
019  private static boolean CHECK_INVARIANTS = PROVIDER.checkInvariants();
020  private static boolean CHECK_API_INVARIANTS = PROVIDER.checkApiInvariants();
021
022  private BrainCheckConfig()
023  {
024  }
025
026  /**
027   * Return true if BrainCheck is running in production mode.
028   *
029   * @return true if BrainCheck is running in production mode.
030   */
031  static boolean isProductionEnvironment()
032  {
033    return PRODUCTION_ENVIRONMENT;
034  }
035
036  /**
037   * Return true if BrainCheck is running in development mode.
038   *
039   * @return true if BrainCheck is running in development mode.
040   */
041  static boolean isDevelopmentEnvironment()
042  {
043    return !isProductionEnvironment();
044  }
045
046  /**
047   * Return true if invariant failures will include a detail message.
048   *
049   * @return true if invariant failures will include a detail message.
050   */
051  public static boolean verboseErrorMessages()
052  {
053    return VERBOSE_ERROR_MESSAGES;
054  }
055
056  /**
057   * Return true if invariants will be checked.
058   *
059   * @return true if invariants will be checked.
060   */
061  public static boolean checkInvariants()
062  {
063    return CHECK_INVARIANTS;
064  }
065
066  /**
067   * Return true if apiInvariants will be checked.
068   *
069   * @return true if apiInvariants will be checked.
070   */
071  public static boolean checkApiInvariants()
072  {
073    return CHECK_API_INVARIANTS;
074  }
075
076  private static final class ConfigProvider
077    extends AbstractConfigProvider
078  {
079    @GwtIncompatible
080    @Override
081    boolean isProductionEnvironment()
082    {
083      return System.getProperty( "braincheck.environment", "production" ).equals( "production" );
084    }
085
086    @GwtIncompatible
087    @Override
088    boolean verboseErrorMessages()
089    {
090      return "true".equals( System.getProperty( "braincheck.verbose_error_messages",
091                                                PRODUCTION_ENVIRONMENT ? "false" : "true" ) );
092    }
093
094    @GwtIncompatible
095    @Override
096    boolean checkInvariants()
097    {
098      return "true".equals( System.getProperty( "braincheck.check_invariants",
099                                                PRODUCTION_ENVIRONMENT ? "false" : "true" ) );
100    }
101
102    @GwtIncompatible
103    @Override
104    boolean checkApiInvariants()
105    {
106      return "true".equals( System.getProperty( "braincheck.check_api_invariants",
107                                                PRODUCTION_ENVIRONMENT ? "false" : "true" ) );
108    }
109  }
110
111  @SuppressWarnings( "unused" )
112  private static abstract class AbstractConfigProvider
113  {
114    boolean isProductionEnvironment()
115    {
116      return "production" == System.getProperty( "braincheck.environment" );
117    }
118
119    boolean verboseErrorMessages()
120    {
121      return "true" == System.getProperty( "braincheck.verbose_error_messages" );
122    }
123
124    boolean checkInvariants()
125    {
126      return "true" == System.getProperty( "braincheck.check_invariants" );
127    }
128
129    boolean checkApiInvariants()
130    {
131      return "true" == System.getProperty( "braincheck.check_api_invariants" );
132    }
133  }
134}
